const generateBtn = document.querySelector("button"); const video = document.querySelector("video"); const downloadBtn = document.querySelectorAll("button")[1]; generateBtn.onclick = async () => { const fileInput = document.querySelector("input[type=file]"); const text = document.querySelectorAll("input")[1].value; if (!fileInput.files[0]) { alert("Upload image first"); return; } const img = document.createElement("img"); img.src = URL.createObjectURL(fileInput.files[0]); await img.decode(); const canvas = document.createElement("canvas"); canvas.width = 720; canvas.height = 1280; const ctx = canvas.getContext("2d"); const stream = canvas.captureStream(30); const recorder = new MediaRecorder(stream); let chunks = []; recorder.ondataavailable = e => chunks.push(e.data); recorder.onstop = () => { const blob = new Blob(chunks,{type:"video/mp4"}); const url = URL.createObjectURL(blob); video.src = url; downloadBtn.onclick = () => { const a = document.createElement("a"); a.href = url; a.download = "ai-reel.mp4"; a.click(); }; }; recorder.start(); let frame = 0; const draw = () => { ctx.fillStyle = "black"; ctx.fillRect(0,0,720,1280); ctx.drawImage(img,0,0,720,900); ctx.fillStyle="white"; ctx.font="60px Arial"; ctx.fillText(text,40,1000); frame++; if(frame < 900){ requestAnimationFrame(draw); }else{ recorder.stop(); } }; draw(); };
